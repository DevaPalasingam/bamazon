bamazonCustomer:

printStock()
	-displays all items
	-calls promptUser()

promptUser()
	-asks user what ID of the product they'd like to buy
	-asks the amount

checkStock()
	-checks if there's enough
		-if yes, print new amount
			-call updateStock()
		-if no, print "Insufficient quantity"
	-call start()

updateStock()
	-updates database to new amount in stock
	-prints total cost of purchase
